        def get_cast(doc)
            cast = []
            doc.xpath('//table[@class="cast"]//tr').each do|v|
                cast.push(Thoth::Character.new do |c|
                    c.actor = v.children[1].content
                    c.name = v.children[3].content
                end)
            end

            cast
        end
