        def get_plot(doc)
            father = nil
            tagline = ''
            doc.xpath('//h5').each do|v|                
                father = v unless v.content != 'Plot:'
            end

            if(!father.nil?)
                father = father.next.next
                tagline = father.children[0].content.strip
            end
            tagline
        end
