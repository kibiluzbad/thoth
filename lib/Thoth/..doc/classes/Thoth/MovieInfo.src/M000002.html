        def get_title_and_year(doc)
            title = ""
            year = ""
            doc.xpath('//h1').each do|v|
                match = /(.[^\(]+)(\(\d+\))/.match(v.content)
                title = match[1].gsub(/\s*/,'') unless match.nil?
                year = match[2].gsub(/[\(\)]/,'') unless match.nil?
            end
            return title, year
        end
