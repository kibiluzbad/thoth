        def get_genres(doc)
            genres = []
            doc.xpath('//div[@class="info"]//div[@class="info-content"]//a').each do|v|
                if(/\/Sections\/Genres\/(\w+)\//.match(v.attr('href')))
                    genres.push(v.content)
                end
            end

            genres
        end
